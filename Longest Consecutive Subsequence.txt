Solution 

class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        int answer=0;
        set<int>s;
        int n=nums.size();
        for(int i=0;i<n;++i)
            s.insert(nums[i]);
        int current=0;
        for(auto p:s)
        {
            int t=p;
            current=1;
            while(s.find(t+current)!=s.end())
            {
                s.erase(s.find(t+current));
                ++current;
            }
            if(current>answer)
                answer=current;
        }
        return answer;
    }
};

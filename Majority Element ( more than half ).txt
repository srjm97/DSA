Solution

class Solution {
public:
    int majorityElement(vector<int>& nums) {
        int answer;
        int count=1,element=nums[0];
        int n=nums.size();
        for(int i=1;i<n;++i)
        {
            if(nums[i]==element)
                ++count;
            else
            {
                --count;
                if(count==0)
                {
                    element=nums[i];
                    count=1;
                }
            }
        }
        answer=element;
        return answer;
    }
};

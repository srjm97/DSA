Solution 

class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int answer=-1000000;
        int n=nums.size();
        int tempsum=0;
        for(int i=0;i<n;++i)
        {
            tempsum+=nums[i];
            if(tempsum>answer)
                answer=tempsum;
            if(tempsum<0)
                tempsum=0;
        }
        return answer;
    }
};
